//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(START)
{
    Looks.SwitchCostumeTo(costume1);
    Forever
    {
        Motion.SetY(0);
        Call c;
    }
}


//----- Custom blocks -------------------------------------------------------------------------------------------------

Define c (warp=true)
{
    Repeat Until Sensing.TouchingObject(bounce1)
    {
        Motion.ChangeYBy(-2);
    }
}
